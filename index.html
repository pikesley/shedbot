    void setup() {
      for (int r = 0; r < count; r++) {
        pinMode(relays[r], OUTPUT);
        pinMode(inputs[r], INPUT);
      }
    }

    void loop() {
      for (int v = 0; v < count; v++) {
        digitalWrite(relays[v], digitalRead(inputs[v]));
      }
